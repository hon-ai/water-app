```ts
const MARK_OPEN: Record<string, (mark: import("prosemirror-model").Mark) => string> = {
  strong: () => "**",
  em: () => "*",
  link: (mark) => `[`,  // open of [text](url); close is special-cased below
};

const MARK_CLOSE: Record<string, (mark: import("prosemirror-model").Mark) => string> = {
  strong: () => "**",
  em: () => "*",
  link: (mark) => `](${mark.attrs.href as string})`,
};

// Marks emit in this nesting order (outermost → innermost).
// Link wraps because [foo](url) is atomic in CommonMark; nesting inside
// causes parsing pain. Strong wraps em because **bold *italic*** is the
// canonical nested-emphasis form.
const MARK_PRIORITY = ["link", "strong", "em"] as const;

function escapeLiterals(text: string): string {
  // Escape CommonMark inline metacharacters so literal `*`, `_`, `[`, `]`,
  // `(`, `)`, `\` in the source text round-trip safely. We're conservative:
  // every potential delimiter is escaped, even when context would parse it
  // literally; the cost is a slightly noisier on-disk format, the win is
  // bulletproof round-trip.
  return text.replace(/([\\*_[\]()])/g, "\\$1");
}

export function inlineToMarkdown(blockNode: import("prosemirror-model").Node): string {
  // Walk the block's inline children. Track the currently-open marks
  // (ordered by MARK_PRIORITY). On each text run:
  //   1. Identify which marks should be open on this run.
  //   2. Diff against currently-open: close the marks that aren't on the
  //      new run (innermost first); open the new marks (outermost first).
  //   3. Emit the run's escaped text.
  // After the last run, close any remaining open marks.
  let out = "";
  const openStack: Array<{ name: string; mark: import("prosemirror-model").Mark }> = [];

  function marksInPriorityOrder(marks: readonly import("prosemirror-model").Mark[]) {
    const byName = new Map<string, import("prosemirror-model").Mark>();
    for (const m of marks) byName.set(m.type.name, m);
    const ordered: Array<{ name: string; mark: import("prosemirror-model").Mark }> = [];
    for (const name of MARK_PRIORITY) {
      const m = byName.get(name);
      if (m) ordered.push({ name, mark: m });
    }
    return ordered;
  }

  function closeUntil(targetSize: number) {
    while (openStack.length > targetSize) {
      const top = openStack.pop()!;
      out += MARK_CLOSE[top.name]!(top.mark);
    }
  }

  blockNode.forEach((child) => {
    if (!child.isText) return;
    const wanted = marksInPriorityOrder(child.marks);
    // Find longest common prefix of openStack and wanted.
    let common = 0;
    while (
      common < openStack.length &&
      common < wanted.length &&
      openStack[common]!.name === wanted[common]!.name &&
      // For link, also require same href; bold/em don't have attrs.
      (openStack[common]!.name !== "link" ||
        openStack[common]!.mark.attrs.href === wanted[common]!.mark.attrs.href)
    ) {
      common += 1;
    }
    // Close anything above the common prefix.
    closeUntil(common);
    // Open the new marks above the common prefix.
    for (let i = common; i < wanted.length; i++) {
      const entry = wanted[i]!;
      out += MARK_OPEN[entry.name]!(entry.mark);
      openStack.push(entry);
    }
    // Emit the text (escaped).
    out += escapeLiterals(child.text ?? "");
  });
  // Close anything still open at the end of the block.
  closeUntil(0);
  return out;
}
```

```ts
type InlineMark = { name: "strong" | "em" | "link"; href?: string };

export function markdownToInlineNodes(
  schema: import("prosemirror-model").Schema,
  line: string,
): import("prosemirror-model").Node[] {
  // Tokenizer state: we walk left-to-right tracking active marks.
  // Recognized syntax (pragmatic CommonMark subset):
  //   **...**       open/close strong
  //   *...*         open/close em (with left/right-flanking rule)
  //   [text](url)   link span (atomic; doesn't nest)
  //   \X            literal X for X ∈ { *, _, [, ], (, ), \ }
  // Anything else is literal text.
  //
  // Emphasis disambiguation: an opening `*` is left-flanking iff the
  // preceding char is whitespace, BOL, or punctuation. A closing `*` is
  // right-flanking iff the following char is whitespace, EOL, or
  // punctuation. We simplify: emphasis delimiters must be adjacent to
  // non-whitespace on the inside and whitespace/punctuation/BOL/EOL on
  // the outside. Intra-word `a*b*c` parses as literal.
  if (line.length === 0) return [];

  type Run = { text: string; marks: InlineMark[] };
  const runs: Run[] = [];
  const active: InlineMark[] = [];
  let buf = "";

  function flush() {
    if (buf.length === 0) return;
    runs.push({ text: buf, marks: [...active] });
    buf = "";
  }

  function isFlankCharOutside(c: string | undefined): boolean {
    // Outside = before an opener or after a closer.
    if (c === undefined) return true; // BOL/EOL
    return /[\s.,;:!?'"()\[\]{}—–-]/.test(c);
  }

  function isFlankCharInside(c: string | undefined): boolean {
    // Inside = just after an opener or just before a closer. Must NOT be whitespace.
    if (c === undefined) return false;
    return /\S/.test(c);
  }

  let i = 0;
  while (i < line.length) {
    const ch = line[i]!;
    const next = line[i + 1];
    const prev = i === 0 ? undefined : line[i - 1];

    // Escape sequence \X → literal X
    if (ch === "\\" && next !== undefined && /[\\*_[\]()]/.test(next)) {
      buf += next;
      i += 2;
      continue;
    }

    // Bold: **
    if (ch === "*" && next === "*") {
      const isOpen = active.every((m) => m.name !== "strong");
      const peekInside = line[i + 2];
      const isClose = !isOpen;
      if (
        isOpen &&
        isFlankCharOutside(prev) &&
        isFlankCharInside(peekInside)
      ) {
        flush();
        active.push({ name: "strong" });
        i += 2;
        continue;
      }
      if (
        isClose &&
        isFlankCharInside(prev) &&
        isFlankCharOutside(peekInside)
      ) {
        flush();
        // Pop strong (it must be the most recent strong; we don't handle
        // crossed/interleaved marks in the pragmatic subset).
        const idx = active.findIndex((m) => m.name === "strong");
        if (idx >= 0) active.splice(idx, 1);
        i += 2;
        continue;
      }
      // Fall through to literal
    }

    // Italic: *
    if (ch === "*" && next !== "*") {
      const isOpen = active.every((m) => m.name !== "em");
      const peekInside = next;
      if (isOpen && isFlankCharOutside(prev) && isFlankCharInside(peekInside)) {
        flush();
        active.push({ name: "em" });
        i += 1;
        continue;
      }
      if (!isOpen && isFlankCharInside(prev) && isFlankCharOutside(peekInside)) {
        flush();
        const idx = active.findIndex((m) => m.name === "em");
        if (idx >= 0) active.splice(idx, 1);
        i += 1;
        continue;
      }
      // Fall through to literal
    }

    // Link: [text](url) — atomic
    if (ch === "[") {
      const closeBracket = line.indexOf("]", i + 1);
      const openParen = closeBracket >= 0 ? line[closeBracket + 1] : undefined;
      if (closeBracket > i + 1 && openParen === "(") {
        const closeParen = line.indexOf(")", closeBracket + 2);
        if (closeParen > closeBracket + 1) {
          const linkText = line.slice(i + 1, closeBracket);
          const href = line.slice(closeBracket + 2, closeParen);
          if (linkText.length > 0 && href.length > 0) {
            flush();
            runs.push({
              text: linkText,
              marks: [...active, { name: "link", href }],
            });
            i = closeParen + 1;
            continue;
          }
        }
      }
      // Fall through to literal
    }

    buf += ch;
    i += 1;
  }
  flush();

  // Convert runs to schema text nodes.
  return runs
    .filter((r) => r.text.length > 0)
    .map((r) => {
      const marks = r.marks.map((m) => {
        if (m.name === "link") return schema.marks.link!.create({ href: m.href ?? "" });
        if (m.name === "strong") return schema.marks.strong!.create();
        return schema.marks.em!.create();
      });
      return schema.text(r.text, marks);
    });
}
```

```tsx
import { useEffect, useState, type CSSProperties } from "react";
import { createPortal } from "react-dom";
import { Bold, Italic, Link2 } from "lucide-react";
import { EditorView } from "prosemirror-view";
import { toggleMark } from "prosemirror-commands";
import { schema } from "./schema";

interface Props {
  editorView: EditorView;
  /** Called when the writer clicks the Link icon. Receives the current
   *  selection's resolved coordinates so the popup can position itself. */
  onLinkClick?: (anchor: { left: number; top: number }) => void;
}

type Rect = { left: number; top: number; width: number; height: number };

function rectFromSelection(view: EditorView): Rect | null {
  const { from, to } = view.state.selection;
  if (from === to) return null;
  const fromCoords = view.coordsAtPos(from);
  const toCoords = view.coordsAtPos(to);
  const left = Math.min(fromCoords.left, toCoords.left);
  const right = Math.max(fromCoords.right, toCoords.right);
  const top = Math.min(fromCoords.top, toCoords.top);
  const bottom = Math.max(fromCoords.bottom, toCoords.bottom);
  return { left, top, width: right - left, height: bottom - top };
}

function selectionHasMark(view: EditorView, markName: "strong" | "em" | "link"): boolean {
  const markType = schema.marks[markName];
  if (!markType) return false;
  const { from, to, empty } = view.state.selection;
  if (empty) return false;
  return view.state.doc.rangeHasMark(from, to, markType);
}

export function SelectionToolbar({ editorView, onLinkClick }: Props) {
  // Mirror selection state into React via a re-render trigger.
  const [version, setVersion] = useState(0);
  const [forceHidden, setForceHidden] = useState(false);

  // Subscribe to editor state changes by re-wrapping dispatchTransaction.
  // The simpler approach: use a tick effect that polls on each animation
  // frame. We use the dispatchTransaction wrapper for precision.
  useEffect(() => {
    const originalDispatch = editorView.props.dispatchTransaction?.bind(editorView);
    editorView.setProps({
      dispatchTransaction(tr) {
        if (originalDispatch) {
          originalDispatch(tr);
        } else {
          editorView.updateState(editorView.state.apply(tr));
        }
        if (tr.selectionSet || tr.docChanged) {
          setVersion((v) => v + 1);
          setForceHidden(false);
        }
      },
    });
    return () => {
      editorView.setProps({ dispatchTransaction: originalDispatch });
    };
  }, [editorView]);

  // Hide on Escape.
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => {
      if (e.key === "Escape") setForceHidden(true);
    };
    window.addEventListener("keydown", onKey);
    return () => window.removeEventListener("keydown", onKey);
  }, []);

  // Hide on scroll (any ancestor scroll changes the selection rect).
  useEffect(() => {
    const onScroll = () => setForceHidden(true);
    window.addEventListener("scroll", onScroll, { capture: true });
    return () => window.removeEventListener("scroll", onScroll, { capture: true } as any);
  }, []);

  // Re-show on next selection change after a hide.
  // (`setForceHidden(false)` already in the dispatch wrapper.)

  // Read from version so the component re-renders on dispatch.
  void version;

  if (forceHidden) return null;
  const rect = rectFromSelection(editorView);
  if (!rect) return null;

  const boldActive = selectionHasMark(editorView, "strong");
  const italicActive = selectionHasMark(editorView, "em");
  const linkActive = selectionHasMark(editorView, "link");

  const toolbarStyle: CSSProperties = {
    position: "fixed",
    // Position 8px above the selection's top, centered horizontally.
    left: rect.left + rect.width / 2,
    top: rect.top - 8,
    transform: "translate(-50%, -100%)",
    display: "flex",
    flexDirection: "row",
    gap: 6,
    padding: "6px 10px",
    borderRadius: "var(--water-r-16)",
    background:
      "color-mix(in oklch, var(--water-hue-coherence) 30%, var(--water-bg-paper))",
    boxShadow:
      "0 0 18px color-mix(in oklch, var(--water-hue-coherence) 50%, transparent)",
    pointerEvents: "auto",
    zIndex: 40,
    animation:
      "water-pill-fade-in var(--water-dur-small) var(--water-ease-out-soft) both",
  };

  const iconButtonStyle = (active: boolean): CSSProperties => ({
    width: 24,
    height: 24,
    display: "inline-flex",
    alignItems: "center",
    justifyContent: "center",
    border: "none",
    background: active
      ? "color-mix(in oklch, var(--water-hue-flow) 40%, transparent)"
      : "transparent",
    color: "var(--water-fg-default)",
    cursor: "pointer",
    borderRadius: "var(--water-r-8)",
  });

  const onMouseDownPreventBlur = (e: React.MouseEvent) => {
    // Don't steal focus from the editor; clicks shouldn't collapse selection.
    e.preventDefault();
  };

  const handleBold = (e: React.MouseEvent) => {
    onMouseDownPreventBlur(e);
    const cmd = toggleMark(schema.marks.strong!);
    cmd(editorView.state, editorView.dispatch);
  };
  const handleItalic = (e: React.MouseEvent) => {
    onMouseDownPreventBlur(e);
    const cmd = toggleMark(schema.marks.em!);
    cmd(editorView.state, editorView.dispatch);
  };
  const handleLink = (e: React.MouseEvent) => {
    onMouseDownPreventBlur(e);
    const r = rectFromSelection(editorView);
    if (!r) return;
    onLinkClick?.({ left: r.left + r.width / 2, top: r.top + r.height + 4 });
  };

  return createPortal(
    <div role="toolbar" aria-label="Text formatting" style={toolbarStyle}>
      <button
        type="button"
        aria-label="Bold"
        data-active={boldActive ? "true" : "false"}
        style={iconButtonStyle(boldActive)}
        onMouseDown={handleBold}
      >
        <Bold size={16} />
      </button>
      <button
        type="button"
        aria-label="Italic"
        data-active={italicActive ? "true" : "false"}
        style={iconButtonStyle(italicActive)}
        onMouseDown={handleItalic}
      >
        <Italic size={16} />
      </button>
      <button
        type="button"
        aria-label="Link"
        data-active={linkActive ? "true" : "false"}
        style={iconButtonStyle(linkActive)}
        onMouseDown={handleLink}
      >
        <Link2 size={16} />
      </button>
    </div>,
    document.body,
  );
}
```

```tsx
import { useEffect, useRef, useState, type CSSProperties } from "react";
import { createPortal } from "react-dom";

interface Props {
  /** Position to anchor the popup at (e.g. just below the selection or toolbar). */
  anchor: { left: number; top: number };
  /** Pre-fill the URL input. Empty string for add mode. */
  initialUrl: string;
  /** True when editing an existing link. Controls whether Remove button is shown. */
  editing: boolean;
  /** Apply the URL as a link mark. Called with the validated URL. */
  onApply: (url: string) => void;
  /** Remove the existing link mark. Called in edit mode when Remove is clicked
   *  OR when Apply is clicked with an empty URL. */
  onRemove: () => void;
  /** Called after Apply / Remove succeed, on Esc, or on outside click. */
  onClose: () => void;
}

function isAcceptable(raw: string): boolean {
  const url = raw.trim();
  if (!url) return false;
  if (/^javascript:/i.test(url)) return false;
  if (/^data:/i.test(url)) return false;
  return true;
}

export function LinkPopup({
  anchor,
  initialUrl,
  editing,
  onApply,
  onRemove,
  onClose,
}: Props) {
  const [value, setValue] = useState(initialUrl);
  const inputRef = useRef<HTMLInputElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    // Autofocus the input on mount.
    inputRef.current?.focus();
    inputRef.current?.select();
  }, []);

  // Close on outside click.
  useEffect(() => {
    const onDown = (e: MouseEvent) => {
      if (!containerRef.current) return;
      if (!containerRef.current.contains(e.target as Node)) {
        onClose();
      }
    };
    window.addEventListener("mousedown", onDown);
    return () => window.removeEventListener("mousedown", onDown);
  }, [onClose]);

  const handleApply = () => {
    const trimmed = value.trim();
    if (trimmed === "" && editing) {
      onRemove();
      onClose();
      return;
    }
    if (!isAcceptable(trimmed)) {
      // Keep popup open; do not close. (Tests check both onApply NOT called
      // and onClose NOT called for rejected URLs.)
      return;
    }
    onApply(trimmed);
    onClose();
  };

  const handleRemove = () => {
    onRemove();
    onClose();
  };

  const containerStyle: CSSProperties = {
    position: "fixed",
    left: anchor.left,
    top: anchor.top,
    transform: "translateX(-50%)",
    display: "flex",
    flexDirection: "row",
    gap: 8,
    alignItems: "center",
    padding: "8px 12px",
    borderRadius: "var(--water-r-16)",
    background:
      "color-mix(in oklch, var(--water-hue-coherence) 30%, var(--water-bg-paper))",
    boxShadow:
      "0 0 18px color-mix(in oklch, var(--water-hue-coherence) 50%, transparent)",
    pointerEvents: "auto",
    zIndex: 41,
    minWidth: 280,
    animation:
      "water-pill-fade-in var(--water-dur-small) var(--water-ease-out-soft) both",
  };

  const inputStyle: CSSProperties = {
    flex: 1,
    border: "none",
    outline: "none",
    background: "transparent",
    color: "var(--water-fg-default)",
    fontFamily: "var(--water-font-sans)",
    fontSize: "var(--water-fs-ui)",
    padding: "4px 6px",
  };

  const buttonStyle: CSSProperties = {
    border: "none",
    background: "transparent",
    color: "var(--water-fg-default)",
    cursor: "pointer",
    padding: "4px 10px",
    borderRadius: "var(--water-r-8)",
    fontFamily: "var(--water-font-sans)",
    fontSize: "var(--water-fs-meta)",
    boxShadow:
      "inset 0 0 0 1px color-mix(in srgb, var(--water-fg-faint) 30%, transparent)",
  };

  return createPortal(
    <div ref={containerRef} role="dialog" aria-label="Link" style={containerStyle}>
      <input
        ref={inputRef}
        aria-label="URL"
        type="url"
        placeholder="https://"
        value={value}
        onChange={(e) => setValue(e.target.value)}
        onKeyDown={(e) => {
          if (e.key === "Enter") {
            e.preventDefault();
            handleApply();
          } else if (e.key === "Escape") {
            e.preventDefault();
            onClose();
          }
        }}
        style={inputStyle}
      />
      <button type="button" onClick={handleApply} style={buttonStyle}>
        Apply
      </button>
      {editing && (
        <button type="button" onClick={handleRemove} style={buttonStyle}>
          Remove
        </button>
      )}
    </div>,
    document.body,
  );
}
```

Inside the `Editor` function body, add state + helpers:

```tsx
type LinkPopupRequest = {
  anchor: { left: number; top: number };
  initialUrl: string;
  editing: boolean;
};

const [linkPopupReq, setLinkPopupRequest] = useState<LinkPopupRequest | null>(null);

const computeAnchorForSelection = (): { left: number; top: number } => {
  const view = viewRef.current;
  if (!view) return { left: 0, top: 0 };
  const { from, to } = view.state.selection;
  const fromCoords = view.coordsAtPos(from);
  const toCoords = view.coordsAtPos(to);
  const left = (fromCoords.left + toCoords.left) / 2;
  // 8px below the selection bottom.
  const top = Math.max(fromCoords.bottom, toCoords.bottom) + 8;
  return { left, top };
};

const getExistingLinkHref = (state: import("prosemirror-state").EditorState): string => {
  const { from } = state.selection;
  const $pos = state.doc.resolve(from);
  const link = $pos.marks().find((m) => m.type.name === "link");
  return (link?.attrs.href as string | undefined) ?? "";
};

const hasLinkUnderCursor = (state: import("prosemirror-state").EditorState): boolean => {
  const $pos = state.doc.resolve(state.selection.from);
  return $pos.marks().some((m) => m.type.name === "link");
};

const handleLinkApply = (url: string) => {
  const view = viewRef.current;
  if (!view) return;
  const { from, to } = view.state.selection;
  const tr = view.state.tr.addMark(from, to, schema.marks.link!.create({ href: url }));
  view.dispatch(tr);
};

const handleLinkRemove = () => {
  const view = viewRef.current;
  if (!view) return;
  const { from, to } = view.state.selection;
  const range = from === to
    ? expandToLinkMark(view.state)
    : { from, to };
  if (!range) return;
  const tr = view.state.tr.removeMark(range.from, range.to, schema.marks.link!);
  view.dispatch(tr);
};

// Expand a collapsed selection to the boundaries of the link mark under cursor.
function expandToLinkMark(state: import("prosemirror-state").EditorState): { from: number; to: number } | null {
  const $pos = state.doc.resolve(state.selection.from);
  const linkMark = $pos.marks().find((m) => m.type.name === "link");
  if (!linkMark) return null;
  // Walk left until the mark stops; walk right until the mark stops.
  let from = state.selection.from;
  let to = state.selection.from;
  const doc = state.doc;
  while (from > 0) {
    const $p = doc.resolve(from - 1);
    if (!$p.marks().some((m) => m.type.name === "link" && m.attrs.href === linkMark.attrs.href)) break;
    from -= 1;
  }
  while (to < doc.content.size) {
    const $p = doc.resolve(to + 1);
    if (!$p.marks().some((m) => m.type.name === "link" && m.attrs.href === linkMark.attrs.href)) break;
    to += 1;
  }
  return { from, to };
}
```

```markdown
## 12. CommonMark emphasis tokenizer is pragmatic, not strict

**What it is.** The M2.5 inline mark parser at `app/src/editor/serialize.ts::markdownToInlineNodes` recognizes `**`, `*`, and `[text](url)` with simplified left-/right-flanking rules. Edge cases like `a*b*c` (intra-word emphasis without surrounding whitespace) parse as literal asterisks instead of `<em>b</em>`.

**Where it lives.** `app/src/editor/serialize.ts::markdownToInlineNodes`. Round-trip property test at `app/src/editor/serialize.test.ts`.

**Why it's fragile.** Strict CommonMark spec is complex (left-flanking vs right-flanking with multiple punctuation classes). The pragmatic subset shipped in M2.5 handles 99% of literary use; the remaining 1% are corner cases mostly seen in code/technical writing, which isn't Water's target.

**What success looks like.** Writers don't encounter parse surprises in normal literary prose. The property test (50 random docs) passes consistently.

**First-look mitigations.**
1. If a writer hits an edge case, suggest wrapping with whitespace or using bold (`**`) instead.
2. Upgrade path: replace `markdownToInlineNodes` with `markdown-it`-based tokenizer or adopt `prosemirror-markdown` library wholesale. ~80 LoC swap.
```
